(function extractUnsolvedProblemIds() {
    // Select all the table rows containing problems
    const rows = document.querySelectorAll('.problems tr:not(.first-row)');
    const unsolvedProblemIds = []; 

    rows.forEach(row => {
        // 1. Check if the row is marked as solved
        // Codeforces usually adds the class 'accepted-problem' to rows of solved problems
        if (row.classList.contains('accepted-problem')) {
            return; // Skip this row
        }

        // 2. Alternatively, check if the status cell contains a checkmark/accepted status
        const statusCell = row.querySelector('td.status-cell');
        if (statusCell && statusCell.querySelector('.solved-accepted')) {
            return; // Skip this row
        }

        // 3. Extract the ID if not solved
        const idCell = row.querySelector('td.id a');
        if (idCell) {
            const rawId = idCell.innerText.trim();
            if (rawId) {
                unsolvedProblemIds.push(rawId);
            }
        }
    });

    if (unsolvedProblemIds.length > 0) {
        console.log(`%c✔ Found ${unsolvedProblemIds.length} unsolved problems:`, "color: #00ff00; font-weight: bold; font-size: 14px;");
        console.log(unsolvedProblemIds.join('\n'));
        return unsolvedProblemIds;
    } else {
        console.log("ℹ No unsolved problems found on this page, or the structure is different.");
    }
})();



(async () => {
    const handle = window.location.pathname.split('/').pop();
    console.log(`Deep-diving into problem stats for ${handle}...`);
    
    try {
        // Fetch basic info
        const infoRes = await fetch(`https://codeforces.com/api/user.info?handles=${handle}`);
        const infoJson = await infoRes.json();
        
        // Fetch contest history
        const ratingRes = await fetch(`https://codeforces.com/api/user.rating?handle=${handle}`);
        const ratingJson = await ratingRes.json();
        
        // Fetch all submissions
        const statusRes = await fetch(`https://codeforces.com/api/user.status?handle=${handle}`);
        const statusJson = await statusRes.json();
        
        if (infoJson.status !== "OK" || ratingJson.status !== "OK" || statusJson.status !== "OK") {
            throw new Error("API returned an error.");
        }

        const user = infoJson.result[0];
        const history = ratingJson.result;
        const submissions = statusJson.result;

        // Process solved problems (verdict: OK) and avoid duplicating same problem solved multiple times
        const solvedProblems = new Set();
        const ratingDistribution = {};
        const tagDistribution = {};
        let totalSubmissions = submissions.length;
        let correctSubmissions = 0;

        submissions.forEach(sub => {
            if (sub.verdict === "OK") {
                correctSubmissions++;
                const prob = sub.problem;
                const probId = `${prob.contestId}${prob.index}`;
                
                if (!solvedProblems.has(probId)) {
                    solvedProblems.add(probId);
                    
                    // Track by difficulty rating
                    if (prob.rating) {
                        ratingDistribution[prob.rating] = (ratingDistribution[prob.rating] || 0) + 1;
                    }
                    
                    // Track by topic tags
                    if (prob.tags && prob.tags.length > 0) {
                        prob.tags.forEach(tag => {
                            tagDistribution[tag] = (tagDistribution[tag] || 0) + 1;
                        });
                    }
                }
            }
        });

        // Sort rating distribution cleanly
        const sortedRatings = Object.keys(ratingDistribution)
            .sort((a, b) => Number(a) - Number(b))
            .reduce((acc, key) => ({ ...acc, [key]: ratingDistribution[key] }), {});

        // Sort tags by frequency
        const sortedTags = Object.entries(tagDistribution)
            .sort((a, b) => b[1] - a[1])
            .reduce((acc, [key, val]) => ({ ...acc, [key]: val }), {});

        const profileSummary = {
            handle: user.handle,
            currentRank: user.rank || "Unrated",
            maxRank: user.maxRank || "Unrated",
            currentRating: user.rating || 0,
            maxRating: user.maxRating || 0,
            metrics: {
                uniqueProblemsSolved: solvedProblems.size,
                submissionAccuracy: totalSubmissions > 0 ? ((correctSubmissions / totalSubmissions) * 100).toFixed(1) + "%" : "0%"
            },
            recentContests: history.slice(-5).map(c => ({
                contest: c.contestName,
                rank: c.rank,
                delta: c.newRating - c.oldRating,
                rating: c.newRating
            })).reverse(),
            solvedCountByDifficulty: sortedRatings,
            topSolvedTags: sortedTags
        };

        console.clear();
        console.log("%c Data extraction complete! Paste this block back into our chat: ", "background: #222; color: #bada55; padding: 5px; font-weight: bold;");
        console.log(JSON.stringify(profileSummary, null, 2));

    } catch (error) {
        console.error("Extraction failed:", error);
    }
})();
